### TS-CE-01

**Name:** `as.data.frame(VarCorr(.))` produces lme4-compatible shape

**Priority:** P0 blocker

**Kind:** parity-vs-lme4

**Dataset / formula:**
- Primary: `sleepstudy` (lme4), `Reaction ~ Days + (1 + Days | Subject)`, REML=TRUE
- Secondary: `Dyestuff` (lme4), `Yield ~ 1 + (1 | Batch)`, REML=TRUE

**Assertion (exact column contract):**

```r
test_that("as.data.frame(VarCorr(.)) produces lme4 grp/var1/var2/vcov/sdcor shape", {
  mm_skip_if_no_lme4()
  env <- new.env(parent = emptyenv())
  utils::data("sleepstudy", package = "lme4", envir = env)
  dat <- env$sleepstudy

  fit <- lmm(Reaction ~ Days + (1 + Days | Subject), data = dat, REML = TRUE,
             control = mm_control(verbose = -1))
  ref <- suppressMessages(suppressWarnings(
    lme4::lmer(Reaction ~ Days + (1 + Days | Subject), data = dat, REML = TRUE)
  ))

  obs <- as.data.frame(VarCorr(fit))
  exp <- as.data.frame(lme4::VarCorr(ref))

  # Required column set
  expect_true(all(c("grp", "var1", "var2", "vcov", "sdcor") %in% names(obs)),
              info = "as.data.frame(VarCorr(.)) must have grp/var1/var2/vcov/sdcor columns")

  # Must contain a Residual row
  expect_true(any(obs$grp == "Residual"),
              info = "as.data.frame(VarCorr(.)) must contain a Residual row")

  # Must contain covariance row (var1 != var2, not NA in both)
  cov_rows <- obs[!is.na(obs$var2), , drop = FALSE]
  expect_true(nrow(cov_rows) >= 1L,
              info = "as.data.frame(VarCorr(.)) must contain at least one covariance row")

  # Row count must equal lme4's (intercept + slope + covariance + residual = 4 for (1+Days|Subject))
  expect_equal(nrow(obs), nrow(exp),
               info = "as.data.frame(VarCorr(.)) row count must match lme4")

  # Numeric parity: vcov and sdcor for diagonal entries vs lme4, tol=1e-3
  diag_obs <- obs[is.na(obs$var2) & obs$grp != "Residual", , drop = FALSE]
  diag_exp <- exp[is.na(exp$var2) & exp$grp != "Residual", , drop = FALSE]
  # Align by grp + var1
  for (i in seq_len(nrow(diag_exp))) {
    hit <- diag_obs$grp == diag_exp$grp[i] & diag_obs$var1 == diag_exp$var1[i]
    expect_true(any(hit),
                info = sprintf("Missing diagonal VarCorr row grp=%s var1=%s",
                               diag_exp$grp[i], diag_exp$var1[i]))
    if (any(hit)) {
      expect_equal(diag_obs$sdcor[hit][1L], diag_exp$sdcor[i],
                   tolerance = 1e-3,
                   info = sprintf("sdcor parity for grp=%s var1=%s", diag_exp$grp[i], diag_exp$var1[i]))
      expect_equal(diag_obs$vcov[hit][1L], diag_exp$vcov[i],
                   tolerance = 1e-3,
                   info = sprintf("vcov parity for grp=%s var1=%s", diag_exp$grp[i], diag_exp$var1[i]))
    }
  }

  # Residual sdcor must match sigma(fit) within 1e-4
  res_sdcor <- obs$sdcor[obs$grp == "Residual"]
  expect_equal(res_sdcor, sigma(fit), tolerance = 1e-6,
               info = "Residual sdcor row must equal sigma(fit)")
  expect_equal(res_sdcor, sigma(ref), tolerance = 1e-4,
               info = "Residual sdcor row must match lme4 sigma within 1e-4")
})
```

**What it guards against:**
- `as.data.frame.mm_varcorr` absent, causing the default list coercion to emit wide
  `table.*`-prefixed columns with no Residual row and no covariance rows — incompatible
  with broom, lattice dotplot, and any downstream tooling that reads the lme4 contract.

**Upstream fixture / engine change needed:** No. The data is already in `$table` and
`$residual_sd`; only an R-side `as.data.frame.mm_varcorr` method is required.

**Priority:** P1

**Kind:** parity-vs-lme4 + error-message

**Dataset / formula:** same as TS-CE-03 (`cbpp`, binomial GLMM)

**Assertion:**

```r
test_that("GLMM residuals(type='pearson') returns values or mm_inference_unavailable", {
  mm_skip_if_no_lme4()
  env <- new.env(parent = emptyenv())
  utils::data("cbpp", package = "lme4", envir = env)
  dat <- env$cbpp

  gfit <- glmm(cbind(incidence, size - incidence) ~ period + (1 | herd),
               data = dat, family = binomial(),
               control = mm_control(verbose = -1))
  gref <- suppressMessages(suppressWarnings(
    lme4::glmer(cbind(incidence, size - incidence) ~ period + (1 | herd),
                data = dat, family = binomial())
  ))

  # Either we get a numeric vector of the right length (parity path),
  # OR we get a typed mm_inference_unavailable error (honest refusal path).
  # A plain "arg should be response" error is NOT acceptable.
  result <- tryCatch(
    residuals(gfit, type = "pearson"),
    mm_inference_unavailable = function(e) e,
    error = function(e) e
  )
  if (inherits(result, "mm_inference_unavailable")) {
    # Honest typed refusal is acceptable while engine support is pending
    expect_true(nchar(conditionMessage(result)) > 0L,
                info = "mm_inference_unavailable must carry a message")
  } else if (inherits(result, "error")) {
    # Any other error (including match.arg failure) is a spec violation
    fail(sprintf("residuals(type='pearson') must not throw a generic error: %s",
                 conditionMessage(result)))
  } else {
    # Numeric path: parity vs lme4 (tol=1e-3)
    expect_equal(length(result), nrow(dat),
                 info = "pearson residuals length must equal nrow(data)")
    expect_equal(as.numeric(result),
                 as.numeric(lme4::residuals(gref, type = "pearson")),
                 tolerance = 1e-3,
                 info = "GLMM pearson residuals parity vs lme4")
  }
})
```

### TS-CE-05

**Name:** `residuals(scaled=TRUE)` is not silently swallowed — either returns scaled residuals or warns/errors

**Priority:** P1

**Kind:** error-message + unit

**Dataset / formula:** `sleepstudy`, `Reaction ~ Days + (1 + Days | Subject)`, REML=TRUE

**Assertion:**

```r
test_that("residuals(scaled=TRUE) does not silently equal unscaled for LMM", {
  mm_skip_if_no_lme4()
  env <- new.env(parent = emptyenv())
  utils::data("sleepstudy", package = "lme4", envir = env)
  dat <- env$sleepstudy
  fit <- lmm(Reaction ~ Days + (1 + Days | Subject), data = dat, REML = TRUE,
             control = mm_control(verbose = -1))

  # The "no silent surgery" principle (CLAUDE.md): if scaled=TRUE is silently ignored
  # and the output is identical to unscaled, the call is fabricating a silent no-op.
  # Acceptable outcomes:
  #   (a) returns residuals / sigma(fit) — genuinely scaled (all values smaller)
  #   (b) signals mm_inference_unavailable with a clear message
  # NOT acceptable: silently returning unscaled residuals without any signal.

  r_plain <- residuals(fit)
  result <- tryCatch(
    residuals(fit, scaled = TRUE),
    mm_inference_unavailable = function(e) e,
    error = function(e) e
  )

  if (inherits(result, "mm_inference_unavailable")) {
    # Typed refusal is acceptable
    expect_true(nchar(conditionMessage(result)) > 0L)
  } else if (inherits(result, "error")) {
    # A hard error is also acceptable (better than silent no-op)
    expect_true(nchar(conditionMessage(result)) > 0L)
  } else {
    # Numeric path: must be divided by sigma (within 1e-12)
    expect_equal(as.numeric(result),
                 as.numeric(r_plain) / sigma(fit),
                 tolerance = 1e-12,
                 info = "residuals(scaled=TRUE) must equal residuals / sigma(fit)")
    # Must differ from unscaled
    expect_false(isTRUE(all.equal(as.numeric(result), as.numeric(r_plain))),
                 info = "residuals(scaled=TRUE) must differ numerically from unscaled")
  }
})
```

**What it guards against:**
- Violation of the "no silent surgery" principle: `scaled = TRUE` captured by `...`
  and silently discarded while returning the same values as the unscaled path, misleading
  the caller without any diagnostic.

**Upstream fixture / engine change needed:** No. The R-side `residuals.mm_lmm` already has
`sigma(object)` available; dividing by it is a pure R change requiring no Rust bridge work.

```r
test_that("as.data.frame(ranef(.)) returns grpvar/term/grp/condval/condsd long frame", {
  mm_skip_if_no_lme4()
  env <- new.env(parent = emptyenv())
  utils::data("sleepstudy", package = "lme4", envir = env)
  dat <- env$sleepstudy

  fit <- lmm(Reaction ~ Days + (1 + Days | Subject), data = dat, REML = TRUE,
             control = mm_control(verbose = -1))
  ref <- suppressMessages(suppressWarnings(
    lme4::lmer(Reaction ~ Days + (1 + Days | Subject), data = dat, REML = TRUE)
  ))

  obs <- as.data.frame(ranef(fit, condVar = TRUE))
  exp <- as.data.frame(lme4::ranef(ref, condVar = TRUE))

  # Required columns (lme4 contract consumed by ggplot2 caterpillar plots)

  for (term_nm in unique(exp$term)) {

    exp_vals <- exp_sub$condval[match(common_grp, exp_sub$grp)]
    expect_equal(obs_vals, exp_vals, tolerance = 1e-2,
                 info = sprintf("condval parity for term '%s'", term_nm))
  }

  # condsd must be finite and positive (derived from postVar diagonal)
  expect_true(all(is.finite(obs$condsd) & obs$condsd >= 0),
              info = "condsd must be non-negative finite (from postVar diagonal)")

  # condsd parity vs lme4 (tol=1e-2)
  for (term_nm in unique(exp$term)) {
    obs_sub <- obs[obs$grpvar == "Subject" & obs$term == term_nm, ]
    exp_sub <- exp[exp$grpvar == "Subject" & exp$term == term_nm, ]
    common_grp <- intersect(obs_sub$grp, exp_sub$grp)
    obs_sd <- obs_sub$condsd[match(common_grp, obs_sub$grp)]
    exp_sd <- exp_sub$condsd[match(common_grp, exp_sub$grp)]
    expect_equal(obs_sd, exp_sd, tolerance = 1e-2,
                 info = sprintf("condsd parity for term '%s'", term_nm))
  }
})
```

```r
test_that("as.data.frame(VarCorr(.)) and as.data.frame(ranef(.)) work after saveRDS/readRDS", {
  mm_skip_if_no_lme4()
  env <- new.env(parent = emptyenv())
  utils::data("sleepstudy", package = "lme4", envir = env)
  dat <- env$sleepstudy

  fit <- lmm(Reaction ~ Days + (1 + Days | Subject), data = dat, REML = TRUE,
             control = mm_control(verbose = -1))

  tmp <- tempfile(fileext = ".rds")
  on.exit(unlink(tmp), add = TRUE)
  saveRDS(fit, tmp)
  fit2 <- readRDS(tmp)

  # VarCorr as.data.frame must work without live Rust handle
  vc1 <- as.data.frame(VarCorr(fit))
  vc2 <- as.data.frame(VarCorr(fit2))
  expect_equal(vc1, vc2,
               info = "as.data.frame(VarCorr(.)) must be identical after revive")

  # ranef as.data.frame must work without live Rust handle
  re1 <- as.data.frame(ranef(fit))
  re2 <- as.data.frame(ranef(fit2))
  expect_equal(re1[, c("grpvar","term","grp","condval")],
               re2[, c("grpvar","term","grp","condval")],
               info = "as.data.frame(ranef(.)) condval must be identical after revive")

  # residuals(type=) must work after revive
  for (tp in c("response", "pearson", "deviance")) {
    expect_equal(residuals(fit, type = tp), residuals(fit2, type = tp),
                 tolerance = 1e-12,
                 info = sprintf("residuals(type='%s') must survive revive", tp))
  }
})
```

**What it guards against:**
- New R-side methods inadvertently calling FFI helpers that require a live Rust handle,
  breaking the save/revive contract that the rest of the package upholds (PRD §8).

**Upstream fixture / engine change needed:** No.
