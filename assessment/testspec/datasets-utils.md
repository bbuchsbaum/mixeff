### TS-DU-01

**Name:** `getME(fit, "Zt")` returns the transpose of the random-effect
design matrix — no error

**Priority:** P0 blocker

**Kind:** unit + parity-vs-lme4

**Dataset / formula:**
- `sleepstudy` (lme4), `Reaction ~ Days + (1 + Days | Subject)`, REML=TRUE

**Assertion:**

```r
test_that("getME(fit, 'Zt') returns a matrix/dgCMatrix without error", {
  mm_skip_if_no_lme4()
  env <- new.env(parent = emptyenv())
  utils::data("sleepstudy", package = "lme4", envir = env)
  dat <- env$sleepstudy

  fit <- lmm(Reaction ~ Days + (1 + Days | Subject), data = dat, REML = TRUE,
             control = mm_control(verbose = -1))

  # Must not error (gap report: currently throws
  #   "Error in t.default(...): argument is not a matrix")
  expect_no_error(Zt <- getME(fit, "Zt"))

  # Must be a matrix-like object (dense or sparse)
  expect_true(is.matrix(Zt) || inherits(Zt, "Matrix"),
              info = "getME(fit,'Zt') must be a matrix or Matrix")

  # Dimensions: nrow = number of RE columns (q), ncol = N observations
  # For (1+Days|Subject) with 18 subjects: q = 2 * 18 = 36, N = 180
  expect_equal(ncol(Zt), nrow(dat),
               info = "Zt must have N columns (one per observation)")
  expect_equal(nrow(Zt), ncol(getME(fit, "Z")),
               info = "nrow(Zt) must equal ncol(Z): Zt == t(Z)")

  # Consistency: Zt == t(Z)
  Z <- getME(fit, "Z")
  expect_equal(as.matrix(Zt), t(as.matrix(Z)),
               tolerance = 1e-12,
               info = "getME(fit,'Zt') must equal t(getME(fit,'Z'))")

  # Parity: same column count as lme4
  ref <- suppressMessages(suppressWarnings(
    lme4::lmer(Reaction ~ Days + (1 + Days | Subject), data = dat, REML = TRUE)
  ))
  ref_Zt <- lme4::getME(ref, "Zt")
  expect_equal(dim(Zt), dim(ref_Zt),
               info = "getME(fit,'Zt') dimensions must match lme4")
})

test_that("getME(fit, 'Zt') works for a random-intercept model", {
  mm_skip_if_no_lme4()
  env <- new.env(parent = emptyenv())
  utils::data("sleepstudy", package = "lme4", envir = env)
  dat <- env$sleepstudy

  fit <- lmm(Reaction ~ Days + (1 | Subject), data = dat, REML = TRUE,
             control = mm_control(verbose = -1))

  expect_no_error(Zt <- getME(fit, "Zt"))

```

**What it guards against:**
- The documented defect: `getME(fit, "Zt")` calls `t(stats::model.matrix(object,
  type = "random"))` but `model.matrix.mm_lmm(type="random")` currently returns
  something that `t.default` cannot transpose (not a plain matrix). Fixing this
  ensures every named component in the `getME` documentation actually works.

**Upstream fixture / engine change needed:** No. The fix is R-side: ensure
`model.matrix.mm_lmm(type = "random")` returns a proper matrix (dense or
sparse), or coerce within `getME.mm_lmm` before calling `t()`.

### TS-DU-02

**Name:** `rePCA(fit)` returns per-component eigenvalue structure for
diagnosing over-parameterized RE

**Priority:** P1

**Kind:** unit + parity-vs-lme4 + error-message

**Dataset / formula:**
- Primary: `sleepstudy`, `Reaction ~ Days + (1 + Days | Subject)`, REML=TRUE
  (non-singular — all components non-zero)
- Secondary: synthetic singular fit — `y ~ 1 + (1 + x | g)` where the slope
  variance is forced to near-zero (boundary fit)

**Assertion:**

```r
test_that("rePCA(fit) returns a list with one element per RE grouping term", {
  mm_skip_if_no_lme4()
  env <- new.env(parent = emptyenv())
  utils::data("sleepstudy", package = "lme4", envir = env)
  dat <- env$sleepstudy

  fit <- lmm(Reaction ~ Days + (1 + Days | Subject), data = dat, REML = TRUE,
             control = mm_control(verbose = -1))

  # rePCA must exist and be callable — it does not exist yet (gap G-DU-02)
  expect_true(existsFunction("rePCA", where = asNamespace("mixeff")),
              info = "rePCA must be exported from mixeff namespace")

  pca_out <- rePCA(fit)

  # Must be a list keyed by grouping factor name
  expect_true(is.list(pca_out),
              info = "rePCA output must be a list")
  expect_true("Subject" %in% names(pca_out),
              info = "rePCA output must be keyed by grouping factor name")

  subj_pca <- pca_out[["Subject"]]

  # Each element must carry eigenvalues (or proportions) as a numeric vector
  # and must be of length equal to the number of RE columns for that term
  # (2 for (1+Days|Subject))
  expect_true(is.numeric(subj_pca$sdev) || is.numeric(subj_pca$eigenvalues),
              info = "rePCA element must carry sdev or eigenvalues")
  eig <- subj_pca$sdev %||% subj_pca$eigenvalues
  expect_equal(length(eig), 2L,
               info = "For (1+Days|Subject) there are 2 RE dimensions")
  expect_true(all(eig >= 0),
              info = "Eigenvalues must be non-negative")
  expect_true(all(is.finite(eig)),
              info = "Eigenvalues must be finite")
})

test_that("rePCA(fit) flags near-zero eigenvalue for a singular fit", {
  # Build a dataset where the slope variance is near-zero (boundary)
  set.seed(42L)
  n <- 40L; g <- 5L
  grp <- factor(rep(seq_len(g), each = n / g))
  x   <- runif(n)
  # All subjects share the same slope — slope variance should collapse to ~0
  y   <- 2 + 1 * x + rep(rnorm(g, sd = 1), each = n / g) + rnorm(n, sd = 0.1)
  dat <- data.frame(y = y, x = x, grp = grp)

  fit <- lmm(y ~ x + (x | grp), data = dat, control = mm_control(verbose = -1))

  # is_singular should agree with rePCA
  expect_true(is.logical(is_singular(fit)),
              info = "is_singular must return a logical")

  pca_out <- rePCA(fit)
  eig <- pca_out[["grp"]]$sdev %||% pca_out[["grp"]]$eigenvalues

  # At least one eigenvalue should be near-zero for the singular fit
  min_eig_pct <- min(eig) / max(eig)
  # We don't assert exact threshold because singular detection is toleranced;
  # we do assert that rePCA produces a non-NULL, non-error result
  expect_true(length(eig) == 2L,
              info = "rePCA must return 2 eigenvalues for (x|grp) term")
})

test_that("rePCA(fit) output is compatible with the lme4 contract shape", {
  mm_skip_if_no_lme4()
  env <- new.env(parent = emptyenv())
  utils::data("sleepstudy", package = "lme4", envir = env)
  dat <- env$sleepstudy

  fit <- lmm(Reaction ~ Days + (1 + Days | Subject), data = dat, REML = TRUE,
             control = mm_control(verbose = -1))
  ref <- suppressMessages(suppressWarnings(
    lme4::lmer(Reaction ~ Days + (1 + Days | Subject), data = dat, REML = TRUE)
  ))

  ref_pca <- lme4::rePCA(ref)
  obs_pca <- rePCA(fit)

  # Same grouping-factor keys
  expect_setequal(names(obs_pca), names(ref_pca),
                  info = "rePCA group names must match lme4")

  # Eigenvalue proportions (normalized) must agree within 1e-2
  ref_eig  <- ref_pca[["Subject"]]$sdev^2
  ref_prop <- ref_eig / sum(ref_eig)
  obs_eig  <- obs_pca[["Subject"]]$sdev %||% obs_pca[["Subject"]]$eigenvalues
  obs_norm <- if (is.null(obs_pca[["Subject"]]$sdev)) obs_eig else obs_eig^2
  obs_prop <- sort(obs_norm / sum(obs_norm), decreasing = TRUE)
  ref_prop_sorted <- sort(ref_prop, decreasing = TRUE)
  expect_equal(obs_prop, ref_prop_sorted, tolerance = 1e-2,
               info = "rePCA eigenvalue proportions must match lme4 within 1e-2")
})
```

**What it guards against:**
- Users diagnosing over-parameterized RE structures getting only a boolean
  (`is_singular()`) instead of the diagnostic richness (component SDs, rotation)
  that lme4's `rePCA` provides. `rePCA` is the Bates et al. 2015 standard tool
  and is explicitly cited in lme4 tutorials.

**Upstream fixture / engine change needed:** Partial. `rePCA` can be implemented
R-side using the `VarCorr` table and Cholesky factorization already available
from the artifact, without a new Rust bridge call. The `sdev` / proportion values
are derived from the fitted covariance matrices stored in `fit$varcorr`. No new
engine changes needed for the basic implementation.

```r
  if (existsFunction("isLMM", where = asNamespace("mixeff"))) {
    expect_true(isLMM(lmm_fit),
                info = "isLMM must return TRUE for mm_lmm")
  }

  if (existsFunction("isGLMM", where = asNamespace("mixeff"))) {
    expect_false(isGLMM(lmm_fit),
                 info = "isGLMM must return FALSE for mm_lmm")

    glmm_fit <- glmm(
      cbind(incidence, size - incidence) ~ period + (1 | herd),
      data = dat_cbpp, family = binomial(),
      control = mm_control(verbose = -1)
    )
    expect_true(isGLMM(glmm_fit),
                info = "isGLMM must return TRUE for mm_glmm")
    if (existsFunction("isLMM", where = asNamespace("mixeff"))) {
      expect_false(isLMM(glmm_fit),
                   info = "isLMM must return FALSE for mm_glmm")
    }
  }

  if (existsFunction("isREML", where = asNamespace("mixeff"))) {
    lmm_reml <- lmm(Reaction ~ Days + (1 | Subject), data = dat_ss,
                    REML = TRUE, control = mm_control(verbose = -1))
    lmm_ml   <- lmm(Reaction ~ Days + (1 | Subject), data = dat_ss,
                    REML = FALSE, control = mm_control(verbose = -1))
    expect_true(isREML(lmm_reml),
                info = "isREML must return TRUE when REML=TRUE")
    expect_false(isREML(lmm_ml),
                 info = "isREML must return FALSE when REML=FALSE")
  } else {
    # Document the gap: class-based workaround must work
    expect_true(inherits(lmm_fit, "mm_lmm"),
                info = "inherits(fit,'mm_lmm') is the current workaround for isLMM")
    skip("isLMM / isGLMM / isREML not yet exported — gap G-DU-12 is open")
  }
})
```

**What it guards against:**

**Upstream fixture / engine change needed:** No. These are pure R-side class

```r
test_that("getME(fit, 'Zt') and getME(fit, 'Z') survive saveRDS/readRDS", {
  mm_skip_if_no_lme4()
  env <- new.env(parent = emptyenv())
  utils::data("sleepstudy", package = "lme4", envir = env)
  dat <- env$sleepstudy

  fit <- lmm(Reaction ~ Days + (1 + Days | Subject), data = dat, REML = TRUE,
             control = mm_control(verbose = -1))

  tmp <- tempfile(fileext = ".rds")
  on.exit(unlink(tmp), add = TRUE)
  saveRDS(fit, tmp)
  fit2 <- readRDS(tmp)

  # Both Z and Zt must work without a live Rust handle
  expect_no_error(Z1  <- getME(fit,  "Z"))
  expect_no_error(Zt1 <- getME(fit,  "Zt"))
  expect_no_error(Z2  <- getME(fit2, "Z"))
  expect_no_error(Zt2 <- getME(fit2, "Zt"))

  expect_equal(as.matrix(Z1),  as.matrix(Z2),  tolerance = 1e-12,
               info = "getME(fit,'Z') must be identical after revive")
  expect_equal(as.matrix(Zt1), as.matrix(Zt2), tolerance = 1e-12,
               info = "getME(fit,'Zt') must be identical after revive")

  # theta, beta, flist, cnms must also survive revive
  for (nm in c("theta", "beta", "flist", "cnms")) {
    expect_no_error(getME(fit2, nm),
                    info = sprintf("getME(fit2, '%s') must work after revive", nm))
  }
})
```

**What it guards against:**
- `getME` components that call Rust FFI helpers (rather than reading from the
  serialized JSON artifact) breaking silently after `saveRDS`/`readRDS`, in

**Upstream fixture / engine change needed:** No. The fix for TS-DU-01 (making
`model.matrix.mm_lmm(type="random")` return a proper matrix from the lazy
