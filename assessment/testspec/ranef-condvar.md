### TS-RC-01

**Name:** `as.data.frame(ranef(., condVar=TRUE))` returns lme4-compatible long frame with `condsd`

**Priority:** P0 blocker

**Kind:** parity-vs-lme4 + unit

**Dataset / formula:**
- Primary: `sleepstudy` (lme4), `Reaction ~ Days + (1 + Days | Subject)`, REML=TRUE
- Secondary: `Dyestuff` (lme4), `Yield ~ 1 + (1 | Batch)`, REML=TRUE
  (intercept-only: single-term, single-group, verifies the simple case does not regress)

**Assertion:**

```r
test_that("as.data.frame(ranef(., condVar=TRUE)) returns grpvar/term/grp/condval/condsd long frame", {
  mm_skip_if_no_lme4()
  env <- new.env(parent = emptyenv())
  utils::data("sleepstudy", package = "lme4", envir = env)
  dat <- env$sleepstudy

  fit <- lmm(Reaction ~ Days + (1 + Days | Subject), data = dat, REML = TRUE,
             control = mm_control(verbose = -1))
  ref <- suppressMessages(suppressWarnings(
    lme4::lmer(Reaction ~ Days + (1 + Days | Subject), data = dat, REML = TRUE)
  ))

  obs <- as.data.frame(ranef(fit, condVar = TRUE))
  exp <- as.data.frame(lme4::ranef(ref, condVar = TRUE))

  # Required columns (lme4 / ggplot2 caterpillar contract)
  required_cols <- c("grpvar", "term", "grp", "condval", "condsd")
  expect_true(
    all(required_cols %in% names(obs)),
    info = sprintf(
      "as.data.frame(ranef(., condVar=TRUE)) must have columns: %s. Got: %s",
      paste(required_cols, collapse = ", "),
      paste(names(obs), collapse = ", ")
    )
  )

  # Row count: 2 terms × 18 subjects = 36
  expect_equal(nrow(obs), nrow(exp),
               info = "row count must match lme4 (2 terms × 18 subjects)")

  # condval parity vs lme4 (tol=1e-2; BLUP values diverge slightly from theta drift)
  for (term_nm in unique(exp$term)) {
    obs_sub <- obs[obs$grpvar == "Subject" & obs$term == term_nm, , drop = FALSE]
    exp_sub <- exp[exp$grpvar == "Subject" & exp$term == term_nm, , drop = FALSE]
    common_grp <- intersect(as.character(obs_sub$grp), as.character(exp_sub$grp))
    expect_true(length(common_grp) > 0L,
                info = sprintf("No matching group levels for term '%s'", term_nm))
    obs_vals <- obs_sub$condval[match(common_grp, as.character(obs_sub$grp))]
    exp_vals <- exp_sub$condval[match(common_grp, as.character(exp_sub$grp))]
    expect_equal(obs_vals, exp_vals, tolerance = 1e-2,
                 info = sprintf("condval parity for term '%s'", term_nm))
  }

  # condsd must be non-negative and finite (derived from sqrt of postVar diagonal)
  expect_true(all(is.finite(obs$condsd) & obs$condsd >= 0),
              info = "condsd must be non-negative finite (sqrt of postVar diagonal)")

  # condsd parity vs lme4 (tol=1e-2)
  for (term_nm in unique(exp$term)) {
    obs_sub <- obs[obs$grpvar == "Subject" & obs$term == term_nm, , drop = FALSE]
    exp_sub <- exp[exp$grpvar == "Subject" & exp$term == term_nm, , drop = FALSE]
    common_grp <- intersect(as.character(obs_sub$grp), as.character(exp_sub$grp))
    obs_sd <- obs_sub$condsd[match(common_grp, as.character(obs_sub$grp))]
    exp_sd <- exp_sub$condsd[match(common_grp, as.character(exp_sub$grp))]
    expect_equal(obs_sd, exp_sd, tolerance = 1e-2,
                 info = sprintf("condsd parity for term '%s'", term_nm))
  }
})

              info = "as.data.frame(ranef(.)) without condVar must have core columns")
  # condsd may be NA or absent; if present, it should be NA
  if ("condsd" %in% names(obs)) {

```

**What it guards against:**
- `as.data.frame.mm_ranef` absent causes dispatch to `base::as.data.frame.list`,
  which returns a wide frame with mangled column names (`Subject..Intercept.`,
  `Subject.Days`), no `grpvar`/`term`/`grp`/`condval`, and no `condsd` — the canonical
  input to lme4 ggplot2 caterpillar recipes is silently broken with no error. The
  `postVar` array that the Rust bridge correctly computed is discarded at the R boundary
  without warning — a "no silent surgery" violation.

**Upstream fixture / engine change needed:** No. The `postVar` array already comes
from the working Rust `wrap__mm_lmm_cond_var_json` bridge. Only an R-side
`as.data.frame.mm_ranef` S3 method is required, extracting `condval` from the data
frame columns and `condsd = sqrt(diag(postVar[,,i]))` for each level.

**Name:** `plot(ranef(.))` dispatches cleanly — no raw base/lattice error

**Priority:** P1

**Kind:** error-message + integration

**Dataset / formula:** `sleepstudy`, `Reaction ~ Days + (1 + Days | Subject)`, REML=TRUE;

```r
test_that("plot(ranef(.)) does not produce raw 'x is a list...' base error", {
  skip_if_not_installed("lattice")
  mm_skip_if_no_lme4()
  env <- new.env(parent = emptyenv())
  utils::data("sleepstudy", package = "lme4", envir = env)
  dat <- env$sleepstudy

  fit <- lmm(Reaction ~ Days + (1 + Days | Subject), data = dat, REML = TRUE,
             control = mm_control(verbose = -1))
  re <- ranef(fit, condVar = TRUE)

  # Currently errors: "'x' is a list, but does not have components 'x' and 'y'"
  # Acceptable outcomes:
  #   (a) returns a trellis / list-of-trellis plot object (full implementation)
  #   (b) signals a typed mm_inference_unavailable with a clear message (honest stub)
  # NOT acceptable: raw base/lattice error with no mixeff context

  result <- tryCatch(
    plot(re),
    mm_inference_unavailable = function(e) e,
    error = function(e) e
  )

  if (inherits(result, "mm_inference_unavailable")) {
    # Typed refusal is acceptable during phased implementation
    expect_true(nchar(conditionMessage(result)) > 0L,
                info = "mm_inference_unavailable must carry a diagnostic message")
    expect_true(grepl("mixeff", conditionMessage(result), ignore.case = TRUE) ||
                  grepl("plot", conditionMessage(result), ignore.case = TRUE),
                info = "message should reference mixeff or plot context")
  } else if (inherits(result, "error")) {
    # Raw error is NOT acceptable — check that it is NOT the base list error
    msg <- conditionMessage(result)
    expect_false(
      grepl("does not have components", msg, fixed = TRUE),
      info = paste(
        "plot(ranef(.)) must not dispatch to base plot giving a raw list error.",
        "Got:", msg
      )
    )
    expect_false(
      grepl("cannot xtfrm", msg, fixed = TRUE),
      info = paste("plot(ranef(.)) must not give lattice xtfrm error. Got:", msg)
    )
  } else {
    # Full implementation: must be a plot object
    expect_true(
      inherits(result, "trellis") || is.list(result),
      info = "plot(ranef(.)) must return a trellis or list of trellis objects"
    )
  }
})
```

**What it guards against:**
- `plot(re)` currently dispatches to `base::plot.default` which errors with
  `'x' is a list, but does not have components 'x' and 'y'`. This is a raw base
  error with zero mixeff context — contrary to the package's "clearer errors" goal.
  Even if full plot support is not yet available, a `plot.mm_ranef` method that issues
  a typed `mm_inference_unavailable` with a helpful message is the minimum bar.

**Upstream fixture / engine change needed:** No. Minimum bar is a `plot.mm_ranef`
stub that emits a typed mixeff error. Full implementation can reuse `dotplot` or
`qqmath` as lme4 does.

### TS-RC-10

**Name:** `postVar` raw variance entries agree with lme4 within absolute tolerance 1e-3 on the balanced balanced design

**Priority:** P1

**Kind:** numerical-tolerance + parity-vs-lme4

**Dataset / formula:** `sleepstudy`, `Reaction ~ Days + (1 + Days | Subject)`, REML=TRUE

**Note:** The gap report shows `max(abs(postVar_mixeff - postVar_lme4)) = 0.0197` (vs
tol 1e-3), driven by the upstream VarCorr/theta drift. The conditional SDs pass at
tol 1e-3. This spec pins BOTH the variance-array tolerance (documented divergence,
currently failing) and the conditional-SD tolerance (currently passing), so that: (a)
any improvement in the optimizer is automatically detected as a new passing assertion,
and (b) regression in the SD path is caught even if the variance path stays failing.
The test is written to *document* the known divergence clearly via `skip` or `expect_gte`
rather than hide it.

**Assertion:**

```r
test_that("postVar conditional SDs agree with lme4 within 1e-3 (passes today)", {
  mm_skip_if_no_lme4()
  env <- new.env(parent = emptyenv())
  utils::data("sleepstudy", package = "lme4", envir = env)
  dat <- env$sleepstudy

  fit <- lmm(Reaction ~ Days + (1 + Days | Subject), data = dat, REML = TRUE,
             control = mm_control(verbose = -1))
  ref <- suppressMessages(suppressWarnings(
    lme4::lmer(Reaction ~ Days + (1 + Days | Subject), data = dat, REML = TRUE)
  ))

  pv_mm   <- attr(ranef(fit, condVar = TRUE)[["Subject"]], "postVar")
  pv_lme4 <- attr(lme4::ranef(ref, condVar = TRUE)[["Subject"]], "postVar")

  # Align by subject name (mixeff has dimnames; add to lme4 for alignment)
  subj_names <- dimnames(pv_mm)[[3L]]
  dimnames(pv_lme4) <- list(
    colnames(lme4::ranef(ref)$Subject),
    colnames(lme4::ranef(ref)$Subject),
    rownames(lme4::ranef(ref)$Subject)
  )
  common_subj <- intersect(subj_names, dimnames(pv_lme4)[[3L]])
  pv_mm_a   <- pv_mm[, , common_subj, drop = FALSE]
  pv_lme4_a <- pv_lme4[, , common_subj, drop = FALSE]

  # Conditional SDs (sqrt of diagonal entries) must be within 1e-3
  for (term_idx in seq_len(dim(pv_mm_a)[[1L]])) {
    csd_mm   <- sqrt(pv_mm_a[term_idx, term_idx, ])
    csd_lme4 <- sqrt(pv_lme4_a[term_idx, term_idx, ])
    expect_equal(csd_mm, csd_lme4, tolerance = 1e-3,
                 info = sprintf(
                   "condSD for term index %d must agree with lme4 within 1e-3 (tol from PRD)",
                   term_idx
                 ))
  }
})

test_that("postVar raw variance entries vs lme4: documents known divergence, pins bar at 5e-2", {
  # This test documents the CURRENT known divergence (max abs diff ~1.97e-2 in postVar[1,1]).
  # The PRD bar is 1e-3 (not yet met for variance entries).
  # This test will PASS at the relaxed 5e-2 tolerance and is explicitly labelled as a
  # known-divergence pin. When the upstream VarCorr/theta drift is resolved, update

  ))

  pv_mm   <- attr(ranef(fit, condVar = TRUE)[["Subject"]], "postVar")
  pv_lme4 <- attr(lme4::ranef(ref, condVar = TRUE)[["Subject"]], "postVar")

  dimnames(pv_lme4) <- list(
    colnames(lme4::ranef(ref)$Subject),

    rownames(lme4::ranef(ref)$Subject)

  max_diff <- max(abs(pv_mm_a - pv_lme4_a))

  # Relaxed bar: current observed diff is ~2e-2; 5e-2 pins us against gross regression
  expect_lte(max_diff, 5e-2,
             label = paste(
               "postVar max abs diff vs lme4.",
               "PRD target is 1e-3 (not yet met; root cause = VarCorr/theta drift).",
               "Current observed:", round(max_diff, 4)
             ))

  # Document the known gap with an informative skip-like label (not actually skipping)
  if (max_diff > 1e-3) {
    message(sprintf(
      paste("[KNOWN DIVERGENCE] postVar max abs diff = %.4e exceeds PRD 1e-3 bar.",

    ))
  }
})
```

**What it guards against:**

- The "known divergence pin" test ensures the gap is documented and will automatically
  become a failing test if the optimizer is improved to meet the 1e-3 bar (because the

  the PR that fixes the optimizer should then tighten both tests to 1e-3).

**Upstream fixture / engine change needed:** The `postVar` variance-entry divergence is
a downstream consequence of the VarCorr/theta drift tracked in

Rust optimizer convergence on the correlated-slope model. No new Rust bridge changes are
needed for the `cond_var()` path itself.
